from typing import TypedDict


class SidebarItem(TypedDict):
    name: str
    icon: str
    badge: int | None


class SidebarState(rx.State):
    sidebar_items: list[SidebarItem] = [
        {"name": "Summary", "icon": "home", "badge": None},
        {"name": "Branch", "icon": "edit", "badge": None},
        {"name": "Course", "icon": "image", "badge": None},
        {"name": "Target", "icon": "file-text", "badge": None},
        {"name": "Marketting", "icon": "message-square", "badge": None},
        # {"name": "Appearance", "icon": "layout", "badge": None},
        # {"name": "Plugins", "icon": "shopping-bag", "badge": None},
        # {"name": "Users", "icon": "users", "badge": None},
        # {"name": "Settings", "icon": "settings", "badge": None},
        # {"name": "Tools", "icon": "wrench", "badge": None},
    ]
    selected_page: str = "Summary"

    @rx.event
    def set_selected_page(self, page: str):
        self.selected_page = page


----------------------------------------

def sidebar_item_component(item: rx.Var[SidebarItem]) -> rx.Component:
    # Each sidebar item (button) is built dynamically using the data in SidebarState.sidebar_items
    # `item` contains: { "icon": ..., "name": ..., "badge": ... }
    return rx.el.button(

        # --- ğŸ”¹ Inside the button: icon, name, optional badge ---
        rx.el.div(
            # ğŸ§© ICON (small logo beside text)
            # h-5 w-5 â†’ icon size; increase for bigger icons
            rx.icon(item["icon"], class_name="h-5 w-5"),

            # ğŸ§© ITEM NAME (the text label)
            # font-medium â†’ medium text weight; can change to font-semibold or text-sm/lg for size
            rx.el.span(item["name"], class_name="font-medium"),

            # ğŸ§© OPTIONAL BADGE (for showing counts, alerts, etc.)
            rx.cond(
                item["badge"],  # if the item has a badge value
                rx.el.span(
                    item["badge"].to_string(),
                    class_name=(
                        # ml-auto â†’ pushes badge to the right
                        # text-xs â†’ small text
                        # bg-blue-500 â†’ blue background (you can change this for each type)
                        # rounded-md â†’ pill-like edges
                        # px / py â†’ padding inside badge (adjust size)
                        "ml-auto text-xs font-semibold bg-blue-500 text-white rounded-md px-1.5 py-0.5"
                    ),
                ),
                rx.fragment(),  # if no badge, render nothing
            ),

            # flex â†’ horizontal layout for icon + text + badge
            # items-center â†’ vertical centering
            # space-x-3 â†’ gap between icon and text
            # w-full â†’ full button width
            class_name="flex items-center space-x-3 w-full",
        ),

        # --- ğŸ”¸ BEHAVIOR WHEN CLICKED ---
        # Calls the state function to set the selected page name
        on_click=lambda: SidebarState.set_selected_page(item["name"]),

        # --- ğŸ¨ BUTTON STYLE (dynamic based on selection) ---
        class_name=rx.cond(
            SidebarState.selected_page == item["name"],

            # âœ… STYLE when this page is selected (active)
            # bg-slate-700 â†’ selected background color (change this to customize highlight)
            # text-white â†’ selected text color
            # px / py â†’ internal padding for button height & width
            # rounded-lg â†’ rounded corners
            "w-full text-left px-3 py-2 rounded-lg bg-slate-700 text-white",

            # âšª STYLE when not selected (normal / hover)
            # text-gray-400 â†’ dimmed text
            # hover:bg-slate-800 â†’ darker background on hover
            # hover:text-white â†’ brighten text on hover
            "w-full text-left px-3 py-2 rounded-lg text-gray-400 hover:bg-slate-800 hover:text-white",
        ),
    )

def sidebar() -> rx.Component:
    return rx.el.aside(
        rx.el.div(

            # --- ğŸ”· HEADER (Logo + Title) SECTION ---
            rx.el.div(
                rx.hstack(
                    # Sidebar Icon (size controlled via h-8 w-8)
                    rx.icon("box", class_name="h-8 w-8 text-white"),
                    # Title text (heading size controls font size)
                    rx.heading("DataMelon", size="6"),
                    # p-5 â†’ padding inside the top header area (around icon + title)
                    #    Increase for more space inside header, decrease for tighter look.
                    class_name="p-5",
                )
            ),

            # --- ğŸ”¹ NAVIGATION ITEMS (Button List) SECTION ---
            rx.el.nav(
                rx.foreach(SidebarState.sidebar_items, sidebar_item_component),
                class_name=(
                    # flex flex-col â†’ stacks items vertically
                    # space-y-4 â†’ vertical gap BETWEEN each sidebar button (Increase for more spacing)
                    # px-2 â†’ left-right padding (space between sidebar border and buttons)
                    "flex flex-col space-y-4 px-2"
                ),
            ),

            # --- ğŸ§± MAIN WRAPPER (Contains Header + Nav) ---
            class_name=(
                # flex flex-col â†’ ensures header and nav are stacked vertically
                # space-y-4 â†’ vertical gap between header section and nav items
                "flex flex-col space-y-4"
            ),
        ),

        # --- ğŸ¨ SIDEBAR CONTAINER (Main Aside Element) ---
        class_name=(
            # w-58 â†’ controls sidebar width (thereâ€™s no Tailwind w-58, use w-56 or w-60 instead)
            # bg-gradient-to-b â†’ creates vertical gradient (topâ†’bottom)
            # from-[#0f172a] â†’ top color
            # to-[#0083b8] â†’ bottom color
            # text-white â†’ makes all text white
            # flex-shrink-0 â†’ prevents sidebar from shrinking when window resizes
            # py-4 â†’ vertical padding inside the sidebar (top and bottom)
            # shadow-[4px_0_12px_rgba(0,0,0,0.5)] â†’ adds right-side shadow (4px offset, blur 12px, darker 0.5 opacity)
            "w-56 bg-gradient-to-b from-[#0f172a] to-[#0083b8] text-white flex-shrink-0 py-4 "
            "shadow-[4px_0_12px_rgba(0,0,0,0.5)]"
        ),
    )
